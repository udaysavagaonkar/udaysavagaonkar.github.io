31 /// An EnclaveAssertionAuthority is an authority for assertions of a particular
32 /// identity type. An EnclaveAssertionAuthority is also identified by its
33 /// authority type. The combination of identity type and authority type uniquely
34 /// identifies an EnclaveAssertionAuthority.
35 ///
36 /// EnclaveAssertionAuthority cannot be instantiated. It is an abstract
37 /// interface that is intended to be extended by subclasses that define a
38 /// particular set of operations on assertion authorities.
39 ///
40 /// See EnclaveAssertionGenerator and EnclaveAssertionVerifier for examples of
41 /// how the EnclaveAssertionAuthority interface can be extended.
42 class EnclaveAssertionAuthority {
43  public:
44  virtual ~EnclaveAssertionAuthority() = default;
45

50  virtual Status Initialize(const std::string& config) = 0;
51 
52  /// Indicates whether this assertion authority has been initialized

56  virtual bool IsInitialized() const = 0;
57

62 
63  /// Gets the type of this assertion authority.
64  ///
65  /// \return The type of this authority.
66  virtual std::string AuthorityType() const = 0;
67 
68  /// Gets a unique identifier for an EnclaveAssertionAuthority with the given
69  /// `identity_type` and `authority_type`.
70  ///
71  /// The identifier is a string that combines `identity_type` and
72  /// `authority_type`. It can be used as a unique identifier for an authority
73  /// that handles assertions for `identity_type` and `authority_type`.
74  ///
75  /// \param identity_type The identity type handled by the authority.
76  /// \param authority_type The authority type of the authority.
77  /// \return The generated authority identifier on success, or a non-OK
78  /// Status on failure.
79  static StatusOr<std::string> GenerateAuthorityId(
80  const EnclaveIdentityType& identity_type, const std::string& authority_type) {
81  std::string serialized;
82  std::vector<std::string> authority_id_tokens = {
83  EnclaveIdentityType_Name(identity_type), authority_type};
84  Status status =
85  asylo::SerializeByteContainers(authority_id_tokens, &serialized);
86  if (!status.ok()) {
87  return status;
88  }
89  return serialized;
90  }
91 
92  protected:
93  /// Indicates whether `description` describes an assertion that is compatible
94  /// with this authority.
95  ///
96  /// This functionality is common to all assertion authorities and is provided
97  /// for convenience of implementing more complex operations.
98  ///
99  /// \param description A description to check for compatibility.
100  /// \return True if `description` is compatible with this authority.
101  bool IsCompatibleAssertionDescription(
102  const AssertionDescription& description) const {
103  return (description.identity_type() == IdentityType()) &&
104  (description.authority_type() == AuthorityType());
105  }
106 };
