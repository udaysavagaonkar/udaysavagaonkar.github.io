28 /// A class for representing either a usable value, or an error.
29 ///
30 /// A StatusOr object either contains a value of type `T` or a Status object
31 /// explaining why such a value is not present. The type `T` must be
32 /// copy-constructible and/or move-constructible.
33 ///
34 /// The state of a StatusOr object may be determined by calling ok() or
35 /// status(). The ok() method returns true if the object contains a valid value.
36 /// The status() method returns the internal Status object. A StatusOr object
37 /// that contains a valid value will return an OK Status for a call to status().
38 ///
39 /// A value of type `T` may be extracted from a StatusOr object through a call
40 /// to ValueOrDie(). This function should only be called if a call to ok()
41 /// returns true. Sample usage:
42 ///
43 /// ```
44 /// asylo::StatusOr<Foo> result = CalculateFoo();
45 /// if (result.ok()) {
46 /// Foo foo = result.ValueOrDie();
47 /// foo->DoSomethingCool();
48 /// } else {
49 /// LOG(ERROR) << result.status();
50 /// }
51 /// ```
52 ///
53 /// If `T` is a move-only type, like `std::unique_ptr<>`, then the value should
54 /// only be extracted after invoking `std::move()` on the StatusOr object.
55 /// Sample usage:
56 ///
57 /// ```
58 /// asylo::StatusOr<std::unique_ptr<Foo>> = CalculateFoo();
59 /// if (result.ok()) {
60 /// std::unique_ptr<Foo> foo = std::move(result).ValueOrDie();
61 /// foo->DoSomethingCool();
62 /// } else {
63 /// LOG(ERROR) << result.status();
64 /// }
65 /// ```
66 ///
67 /// StatusOr is provided for the convenience of implementing functions that
68 /// return some value but may fail during execution. For instance, consider a
69 /// function with the following signature:
70 ///
71 /// ```
72 /// asylo::Status CalculateFoo(int *output);
73 /// ```
74 ///
75 /// This function may instead be written as:
76 ///
77 /// ```
78 /// asylo::StatusOr<int> CalculateFoo();
79 /// ```
80 template <class T>
81 class StatusOr {
82  public:
83  /// Constructs a StatusOr object that contains a non-OK status.
84  /// The non-OK status has an error code of -1. This is a non-standard POSIX
85  /// error code and is used in this context to indicate an unknown error.
86  ///
87  /// This constructor is marked `explicit` to prevent attempts to `return {}`
88  /// from a function with a return type of, for example,
89  /// `StatusOr<std::vector<int>>`. While `return {}` seems like it would return
90  /// an empty vector, it will actually invoke the default constructor of
91  /// StatusOr.
92  explicit StatusOr()
93  : variant_(Status(error::GoogleError::UNKNOWN, "Unknown error")),
94  has_value_(false) {}
95 
96  ~StatusOr() {
97  if (has_value_) {
98  variant_.value_.~T();
99  } else {
100  variant_.status_.~Status();
101  }
102  }
103 
104  /// Constructs a StatusOr object with the given non-OK Status object. All
105  /// calls to ValueOrDie() on this object will abort. The given `status` must
106  /// not be an OK status, otherwise this constructor will abort.
107  ///
108  /// This constructor is not declared explicit so that a function with a return
109  /// type of `StatusOr<T>` can return a Status object, and the status will be
110  /// implicitly converted to the appropriate return type as a matter of
111  /// convenience.
112  ///
113  /// \param status The non-OK Status object to initalize to.
114  StatusOr(const Status &status) : variant_(status), has_value_(false) {
115  if (status.ok()) {
116  LOG(FATAL) << "Cannot instantiate StatusOr with Status::OkStatus()";
117  }
118  }
119 
120  /// Constructs a StatusOr object containing `value`. The resulting object is
121  /// considered to have an OK status. The wrapped element can be accessed with
122  /// ValueOrDie().
123  ///
124  /// This constructor is not declared `explicit` so that a function with a
125  /// return type of `StatusOr<T>` can return an element of type `T`, and this
126  /// value will be implicitly converted to the appropriate return type as
127  /// matter of convenience.
128  ///
129  /// \param value The wrapped value to initialize to.
130  StatusOr(const T &value)
131  : variant_(value), has_value_(true) {}
132 
133  /// Constructs a StatusOr object that contains `value`. The resulting object
134  /// is considered to have an OK status. The wrapped element can be accessed
135  /// with ValueOrDie().
136  ///
137  /// This constructor is not declared explicit so that a function with a return
138  /// type of `StatusOr<T>` can return an element of type `T &&`, and this value
139  /// will be implicitly converted to the appropriate return type as a matter of
140  /// convenience.
141  ///
142  /// \param value The value to move-initialize to.
143  StatusOr(T &&value)
144  : variant_(std::move(value)), has_value_(true) {}
145 
146  /// Copy constructor.
147  StatusOr(const StatusOr &other) : has_value_(other.has_value_) {
148  if (has_value_) {
149  new (&variant_) variant(other.variant_.value_);
150  } else {
151  new (&variant_) variant(other.variant_.status_);
152  }
153  }
154 
155  /// Copy-assignment operator.
156  ///
157  /// \param other The StatusOr object to copy.
158  StatusOr &operator=(const StatusOr &other) {
159  // Check for self-assignment.
160  if (this == &other) {
161  return *this;
162  }
163 
164  // Construct the variant object using the variant object of the source.
165  if (other.has_value_) {
166  AssignValue(other.variant_.value_);
167  } else {
168  AssignStatus(other.variant_.status_);
169  }
170  return *this;
171  }
172 
173  /// Move constructor.
174  ///
175  /// Sets `other` to contain a non-OK status with a `StatusError::INVALID`
176  /// error code.
177  ///
178  /// \param other The StatusOr object to copy and set to a non-OK status.
179  StatusOr(StatusOr &&other) : has_value_(other.has_value_) {
180  if (has_value_) {
181  new (&variant_) variant(std::move(other.variant_.value_));
182  } else {
183  new (&variant_) variant(std::move(other.variant_.status_));
184  }
185 
186  // The donor object may have previously held a valid value that is now
187  // inaccessible. Clear its current state.
188  other.Clear();
189  }
190 
191  /// Move-assignment operator.
192  ///
193  /// Sets `other` to contain a non-OK status with a `StatusError::INVALID`
194  /// error code.
195  ///
196  /// \param other The StatusOr object to assign from and set to a non-OK
197  /// status.
198  StatusOr &operator=(StatusOr &&other) {
199  // Check for self-assignment.
200  if (this == &other) {
201  return *this;
202  }
203 
204  // Construct the variant object using the variant object of the donor.
205  if (other.has_value_) {
206  AssignValue(std::move(other.variant_.value_));
207  } else {
208  AssignStatus(std::move(other.variant_.status_));
209  }
210 
211  // The donor object may have previously held a valid value that is now
212  // inaccessible. Clear its current state.
213  other.Clear();
214 
215  return *this;
216  }
217 
218  // operators.
219 
220  /// Indicates whether the object contains a `T` value.
221  ///
222  /// \return True if this StatusOr object's status is OK (i.e. a call to ok()
223  /// returns true). If this function returns true, then it is safe to access
224  /// the wrapped element through a call to ValueOrDie().
225  bool ok() const {
226  return has_value_;
227  }
228 
229  /// Gets the stored status object, or an OK status if a `T` value is stored.
230  ///
231  /// \return The stored non-OK status object, or an OK status if this object
232  /// has a value.
233  Status status() const {
234  return ok() ? Status::OkStatus() : variant_.status_;
235  }
236 
237  /// Gets the stored `T` value.
238  ///
239  /// This method should only be called if this StatusOr object's status is OK
240  /// (i.e. a call to ok() returns true), otherwise this call will abort.
241  ///
242  /// \return The stored `T` value.
243  const T &ValueOrDie() const & {
244  if (!ok()) {
245  LOG(FATAL) << "Object does not have a usable value";
246  }
247  return variant_.value_;
248  }
249 
250  /// Gets a mutable reference to the stored `T` value.
251  ///
252  /// This method should only be called if this StatusOr object's status is OK
253  /// (i.e. a call to ok() returns true), otherwise this call will abort.
254  ///
255  /// \return The stored `T` value.
256  T &ValueOrDie() & {
257  if (!ok()) {
258  LOG(FATAL) << "Object does not have a usable value";
259  }
260  return variant_.value_;
261  }
262 
263  /// Moves and returns the internally-stored `T` value.
264  ///
265  /// This method should only be called if this StatusOr object's status is OK
266  /// (i.e. a call to ok() returns true), otherwise this call will abort. The
267  /// StatusOr object is invalidated after this call and will be updated to
268  /// contain a non-OK status with a `StatusError::INVALID` error code.
269  ///
270  /// \return The stored `T` value.
271  T ValueOrDie() && {
272  if (!ok()) {
273  LOG(FATAL) << "Object does not have a usable value";
274  }
275  T tmp(std::move(variant_.value_));
276 
277  // Invalidate this StatusOr object.
278  Clear();
279  return std::move(tmp);
280  }
281 
282  private:
283  // Clears the current state of the StatusOr object and sets it to contain a
284  // Status object with a StatusError::INVALID error code.
285  void Clear() {
286  AssignStatus(Status(error::StatusError::INVALID, "The object was moved"));
287  }
288 
289  // Resets the |variant_| member to contain the |status| and sets |has_value_|
290  // to indicate that the StatusOr object does not have a value. Destroys the
291  // existing |variant_| member if necessary.
292  template <class U>
293  void AssignStatus(U &&status) {
294  if (ok()) {
295  variant_.value_.~T();
296  new (&variant_) variant(std::forward<U>(status));
297  } else {
298  // Reuse the existing Status object.
299  variant_.status_ = std::forward<U>(status);
300  }
301  has_value_ = false;
302  }
303 
304  // Resets the |variant_| member to contain the |value| and sets |has_value_|
305  // to indicate that the StatusOr object has a value. Destroys the existing
306  // |variant_| member.
307  template <class U>
308  void AssignValue(U &&value) {
309  if (ok()) {
310  // We cannot assume that T is move-assignable.
311  variant_.value_.~T();
312  } else {
313  variant_.status_.~Status();
314  }
315  new (&variant_) variant(std::forward<U>(value));
316  has_value_ = true;
317  }
318 
319  union variant {
320  // A non-OK status.
321  Status status_;
322 
323  // An element of type T.
324  T value_;
325 
326  variant() {}
327 
328  variant(const Status &status) : status_(status) {}
329 
330  variant(Status &&status) : status_(std::move(status)) {}
331 
332  variant(const T &value) : value_(value) {}
333 
334  variant(T &&value) : value_(std::move(value)) {}
335 
336  // This destructor must be explicitly defined because it is deleted due to
337  // the variant type having non-static data members with non-trivial
338  // destructors.
339  ~variant() {}
340  };
341 
342  // One of: a non-OK status or an element of type T.
343  variant variant_;
344 
345  // Indicates the active member of the variant_ member.
346  //
347  // A value of true indicates that value_ is the active member of variant_.
348  //
349  // A value of false indicates that status_ is the active member of variant_.
350  bool has_value_;
351 };
