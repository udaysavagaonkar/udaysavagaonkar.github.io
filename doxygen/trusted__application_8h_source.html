33 /// Abstract base class for trusted applications.
34 ///
35 /// To implement an enclave application, client code declares a
36 /// TrustedApplication and implements the entry points it wishes to handle. For
37 /// example:
38 ///
39 /// ```
40 /// class HelloWorld : public TrustedApplication {
41 /// public:
42 /// Status Initialize(const EnclaveConfig &config) override {
43 /// enc_untrusted_puts("Hello!");
44 /// return Status::OkStatus();
45 /// }
46 ///
47 /// Status Run(const EnclaveInput &input, EnclaveOutput *output) override {
48 /// enc_untrusted_puts("Running!");
49 /// return Status::OkStatus();
50 /// }
51 ///
52 /// Status Finalize(const EnclaveFinal &fini) override {
53 /// enc_untrusted_puts("Goodbye!");
54 /// return Status::OkStatus();
55 /// }
56 /// };
57 /// ```
58 ///
59 /// At startup, the runtime will call the user supplied function
60 /// BuildTrustedApplication and install the returned instance as the handler for
61 /// enclave entries events. For instance:
62 ///
63 /// ```
64 /// TrustedApplication *BuildTrustedApplication() {
65 /// return new HelloWorld;
66 /// }
67 /// ```
68 ///
69 /// Note that types derived from TrustedApplication must be trivially
70 /// destructible, and any such destructor will never be invoked by the runtime.
71 class TrustedApplication {
72  public:
73  /// An enumeration of possible enclave runtime states.
74  enum class State {
75  kUninitialized,
76  kInitializing,
77  kRunning,
78  kFinalizing,
79  kFinalized,
80  };
81 
82  /// \private
83  Status InitializeInternal(const EnclaveConfig &config);
84 
85  /// Implements enclave initialization entry-point.
86  ///
87  /// \param config The configuration used to initialize the enclave.
88  /// \return An OK status or an error if the enclave could not be initialized.
89  virtual Status Initialize(const EnclaveConfig &config) {
90  return Status::OkStatus();
91  }
92 
93  /// Implements enclave execution entry-point.
94  ///
95  /// \param input Message passed to determine behavior for the Run routine.
96  /// \param output Message passed back to the untrusted caller.
97  /// \return OK status or error
98  virtual Status Run(const EnclaveInput &input, EnclaveOutput *output) {
99  return Status::OkStatus();
100  }
101 
102  /// Implements enclave finalization behavior.
103  ///
104  /// \param final_input Message passed on enclave finalization.
105  /// \return OK status or error
106  virtual Status Finalize(const EnclaveFinal &final_input) {
107  return Status::OkStatus();
108  }
109 
110  /// Trivial destructor.
111  ///
112  /// Trivial destructor. Note that classes derived from of TrustedApplication
113  /// must not add a non-trivial destructor, as they will not be called by the
114  /// enclave runtime.
115  virtual ~TrustedApplication() = default;
116 
117  /// Returns the enclave state in a thread-safe manner.
118  State GetState() LOCKS_EXCLUDED(mutex_);
119 
120  private:
121  // Tracks the current enclave state.
122  State enclave_state_ = State::kUninitialized GUARDED_BY(mutex_);
123  absl::Mutex mutex_;
124 
125  // Verifies the expected enclave state and sets a new one in thread-safe
126  // manner. Returns error if the verification fails.
127  asylo::Status VerifyAndSetState(const State &expected_state,
128  const State &new_state)
129  LOCKS_EXCLUDED(mutex_);
130 
131  // Sets the enclave state in thread-safe manner.
132  void SetState(const State &state) LOCKS_EXCLUDED(mutex_);
133 
134  friend int __asylo_user_init(const char *name, const char *config,
135  size_t config_len, char **output,
136  size_t *output_len);
137  friend int __asylo_user_run(const char *input, size_t input_len,
138  char **output, size_t *output_len);
139  friend int __asylo_user_fini(const char *input, size_t input_len,
140  char **output, size_t *output_len);
141  friend int __asylo_threading_donate();
142  friend int __asylo_handle_signal(const char *input, size_t input_len);
143 };
