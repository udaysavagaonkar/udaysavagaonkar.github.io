47 /// Creates a message and logs it to file.
48 ///
49 /// `LOG(severity)` returns a stream object that can be written to with the `<<`
50 /// operator. Log messages are emitted with terminating newlines.
51 /// Example:
52 ///
53 /// ```
54 /// LOG(INFO) << "Found" << num_cookies << " cookies";
55 /// ```
56 ///
57 /// \param severity The severity of the log message, one of `LogSeverity`. The
58 /// FATAL severity will terminate the program after the log is emitted.
59 #define LOG(severity) COMPACT_ASYLO_LOG_##severity.stream()
60 
61 /// A command to LOG only if a condition is true. If the condition is false,
62 /// nothing is logged.
63 /// Example:
64 ///
65 /// ```
66 /// LOG_IF(INFO, num_cookies > 10) << "Got lots of cookies";
67 /// ```
68 ///
69 /// \param severity The severity of the log message, one of `LogSeverity`. The
70 /// FATAL severity will terminate the program after the log is emitted.
71 /// \param condition The condition that determines whether to log the message.
72 #define LOG_IF(severity, condition)
73  !(condition) ? (void)0 : asylo::LogMessageVoidify() & LOG(severity)
74 
75 /// A `LOG` command with an associated verbosity level. The verbosity threshold
76 /// may be configured at runtime with `set_vlog_level` and `InitLogging`.
77 ///
78 /// `VLOG` statements are logged at `INFO` severity if they are logged at all.
79 /// The numeric levels are on a different scale than the severity levels.
80 /// Example:
81 ///
82 /// ```
83 /// VLOG(1) << "Print when VLOG level is set to be 1 or higher";
84 /// ```
85 ///
86 /// \param level The numeric level that determines whether to log the message.
87 #define VLOG(level) LOG_IF(INFO, (level) <= get_vlog_level())
88 
89 /// Terminates the program with a fatal error if the specified condition is
90 /// false.
91 ///
92 /// Example:
93 ///
94 /// ```
95 /// CHECK(!cheese.empty()) << "Out of Cheese";
96 /// ```
97 ///
98 /// Might produce a message like:
99 ///
100 /// `Check_failed: !cheese.empty() Out of Cheese`
101 #define CHECK(condition)
102  LOG_IF(FATAL, !(condition)) << "Check failed: " #condition " "

165 /// \cond Internal
166 template <typename T1, typename T2>
167 std::string *MakeCheckOpString(const T1 &v1, const T2 &v2, const char *exprtext) {
168  CheckOpMessageBuilder comb(exprtext);
169  MakeCheckOpValueString(comb.ForVar1(), v1);
170  MakeCheckOpValueString(comb.ForVar2(), v2);
171  return comb.NewString();
172 }

174 // Helper functions for CHECK_OP macro.
175 // The (int, int) specialization works around the issue that the compiler
176 // will not instantiate the template version of the function on values of
177 // unnamed enum type - see comment below.
178 //
179 /// \param name An identifier that is the name of the comparison, such as
180 /// Check_EQ or Check_NE.
181 /// \param op The comparison operator, such as == or !=.
182 #define DEFINE_CHECK_OP_IMPL(name, op)
183  template <typename T1, typename T2>
184  inline std::string *name##Impl(const T1 &v1, const T2 &v2,
185  const char *exprtext) {
186  if (ABSL_PREDICT_TRUE(v1 op v2)) return nullptr;
187  return MakeCheckOpString(v1, v2, exprtext);
188  }
189  inline std::string *name##Impl(int v1, int v2, const char *exprtext) {
190  return name##Impl<int, int>(v1, v2, exprtext);
191  }

226 /// Compares `val1` and `val2` with `op`, and does `log` if false.
227 ///
228 /// \param name An identifier that is the name of the comparison, such as
229 /// Check_EQ or Check_NE.
230 /// \param op The comparison operator, such as == or !=.
231 /// \param val1 The first variable to be compared.
232 /// \param val2 The second variable to be compared.
233 /// \param log The log action to be performed if the comparison returns false.
234 #define CHECK_OP_LOG(name, op, val1, val2, log)
235  while (std::unique_ptr<std::string> _result = std::unique_ptr<std::string>(name##Impl(
236  GetReferenceableValue(val1), GetReferenceableValue(val2),
237  #val1 " " #op " " #val2)))
238  log(__FILE__, __LINE__, *_result).stream()
239 
240 /// Compares `val1` and `val2` with `op`, and produces a LOG(FATAL) if false.
241 ///
242 /// \param name An identifier that is the name of the comparison, such as
243 /// Check_EQ or Check_NE.
244 /// \param op The comparison operator, such as == or !=.
245 /// \param val1 The first variable to be compared.
246 /// \param val2 The second variable to be compared.
247 #define CHECK_OP(name, op, val1, val2)
248  CHECK_OP_LOG(name, op, val1, val2, asylo::LogMessageFatal)
249 
250 /// Produces a LOG(FATAL) unless `val1` equals `val2`.
251 #define CHECK_EQ(val1, val2) CHECK_OP(Check_EQ, ==, val1, val2)
252 /// Produces a LOG(FATAL) unless `val1` does not equal to `val2`.
253 #define CHECK_NE(val1, val2) CHECK_OP(Check_NE, !=, val1, val2)
254 /// Produces a LOG(FATAL) unless `val1` is less than or equal to `val2`.
255 #define CHECK_LE(val1, val2) CHECK_OP(Check_LE, <=, val1, val2)
256 /// Produces a LOG(FATAL) unless `val1` is less than `val2`.
257 #define CHECK_LT(val1, val2) CHECK_OP(Check_LT, <, val1, val2)
258 /// Produces a LOG(FATAL) unless `val1` is greater than or equal to `val2`.
259 #define CHECK_GE(val1, val2) CHECK_OP(Check_GE, >=, val1, val2)
260 /// Produces a LOG(FATAL) unless `val1` is greater than `val2`.
261 #define CHECK_GT(val1, val2) CHECK_OP(Check_GT, >, val1, val2)

263 /// Checks that the argument is not null, and returns it.
264 ///
265 /// Unlike other `CHECK` macros, this one returns its input, so it can be used
266 /// in initializers. Outside initializers, prefer `CHECK`.
267 ///
268 /// `CHECK_NOTNULL` works for both raw pointers and (compatible) smart pointers
269 /// including `std::unique_ptr` and `std::shared_ptr`.
270 ///
271 /// For smart pointers `CHECK_NOTNULL` returns a reference to its argument,
272 /// preserving the value category (i.e., an rvalue reference for an
273 /// rvalue argument, and an lvalue reference otherwise). For pre-C++11
274 /// compilers that's not possible, so as a best available approximation
275 /// a reference-to-const will be returned if the argument is an rvalue.
276 ///
277 /// \param val The value being compared.
278 #define CHECK_NOTNULL(val)
279  CheckNotNull(__FILE__, __LINE__, "'" #val "' Must be non NULL", (val))

399 /// Logs a message if the given value of type `T` is null, and then forwards the
400 /// value.
401 ///
402 /// In C++11, all cases can be handled by a single function. Since the value
403 /// category of the argument is preserved (also for rvalue references),
404 /// member initializer lists like the one below will compile correctly:
405 ///
406 /// ```
407 /// Foo()
408 /// : x_(CHECK_NOTNULL(MethodReturningUniquePtr())) {}
409 /// ```
410 ///
411 /// \param file The source file that produced the log.
412 /// \param line The source code line that produced the log.
413 /// \param exprtext A string representation of the code in `file` at `line`.
414 /// \param t The parameter being checked for null.
415 template <typename T>
416 T CheckNotNull(const char *file, int line, const char *exprtext, T &&t) {
417  if (ABSL_PREDICT_FALSE(!t)) {
418  LogMessage(file, line, std::string(exprtext));
419  }
420  return std::forward<T>(t);
421 }
